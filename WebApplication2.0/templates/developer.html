Step 2: Making Your First API Request

Here's how you can make a request using Python: 


import requests

api_endpoint = "https://api.deeptracers.com/v1/detect"
api_key = "sk-12345abcde67890ghijklmnopqrstuv"

headers = {
    "Authorization": f"Bearer {api_key}",
    "Content-Type": "application/json"
}

    "media_url": "https://example.com/media/sample-video.mp4"
}

response = requests.post(api_endpoint, headers=headers, json=data)
print(response.json())

Step 4: Adding to Your Application

Hereâ€™s an example of integrating it into a Flask web app:


from flask import Flask, request, jsonify
import requests

app = Flask(__name__)

API_KEY = "sk-12345abcde67890ghijklmnopqrstuv"
API_URL = "https://api.deeptracers.com/v1/detect"

@app.route('/analyze', methods=['POST'])
def analyze_media():
    media_url = request.json.get('media_url')
    headers = {
        "Authorization": f"Bearer {API_KEY}",
        "Content-Type": "application/json"
    }
    data = {"media_url": media_url}
    response = requests.post(API_URL, headers=headers, json=data)
    return jsonify(response.json())

if __name__ == '__main__':
    app.run(debug=True)
